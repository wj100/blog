textarea多行邮箱格式校验

效果

代码 vue element

const validMail = (rule, value, callback) => {
      if (!value) {
        callback(new Error('请输入邮箱'));
      }
      value = value.replace(/[\n]/g, ',');
      const reg = RegExp(/^(?=.*\.)(?=.*@).*$/);
      if (!value.includes(',')) {
        if (!reg.test(value)) {
          callback(new Error('邮箱格式错误'));
        } else {
          callback();
        }
      } else {
        let flag = -1;
        value.split(',').map((mail, index) => {
          if (mail && !reg.test(mail)) {
            flag = index;
          }
        });
        if (flag !== -1) {
          callback(new Error(`第${flag + 1}行邮箱格式错误`));
        } else {
          callback();
        }
      }
    };
